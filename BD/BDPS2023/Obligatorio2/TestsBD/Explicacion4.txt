SELECT c.nombre AS NombreComunidad,
u.nombre AS NombreUsuario,
COUNT(v.id) AS CantidadVideos
FROM Comunidad c
INNER JOIN Pertenece p ON C.nombre = p.nombre
INNER JOIN Usuario u ON P.idUsuario = u.id
INNER JOIN Video v ON U.id = v.idUsuario
WHERE c.nombre IN (
 SELECT c2.nombre
 FROM Comunidad c2
 INNER JOIN Pertenece p2 ON c2.nombre = p2.nombre
 INNER JOIN Video v2 ON p2.idUsuario = v2.idUsuario
 GROUP BY c2.nombre
 HAVING COUNT(v2.id) >= 10
)
GROUP BY c.nombre, u.nombre
HAVING COUNT(v.id) in (
 SELECT MAX(cant_vid)
 FROM (
 SELECT c3.nombre AS ComunidadNombre,
u3.nombre AS UsuarioNombre,
COUNT(v3.id) AS cant_vid
 FROM Comunidad c3
 INNER JOIN Pertenece p3 ON c3.nombre = p3.nombre
 INNER JOIN Usuario u3 ON p3.idUsuario = u3.id
 INNER JOIN Video v3 ON u3.id = v3.idUsuario
 GROUP BY c3.nombre, u3.nombre
 ) AS cant_videos_usuarios
 WHERE cant_videos_usuarios.ComunidadNombre = c.nombre
)
ORDER BY c.nombre;

SELECT MAX(cant_vid), c.nombre
    FROM comunidad c,(
        SELECT c3.nombre AS ComunidadNombre,u3.nombre AS UsuarioNombre, COUNT(v3.id) AS cant_vid
        FROM Comunidad c3
        INNER JOIN Pertenece p3 ON c3.nombre = p3.nombre
        INNER JOIN Usuario u3 ON p3.idUsuario = u3.id
        INNER JOIN Video v3 ON u3.id = v3.idUsuario
        GROUP BY c3.nombre, u3.nombre
    ) AS cant_videos_usuarios
    WHERE cant_videos_usuarios.ComunidadNombre = c.nombre
    GROUP BY c.nombre;
